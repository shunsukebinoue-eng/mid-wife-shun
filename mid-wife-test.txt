// --- 2. LOGIC ---

    let currentQuestions = [];
    let currentQuestionIndex = 0;
    let score = 0;
    let userMistakes = [];
    const MAX_QUESTIONS = 15; // 出題数を15問に設定

    // Elements
    const startScreen = document.getElementById('start-screen');
    const quizScreen = document.getElementById('quiz-screen');
    const resultScreen = document.getElementById('result-screen');
    const questionText = document.getElementById('q-text');
    const genreBadge = document.getElementById('q-genre');
    const diffBadge = document.getElementById('q-diff');
    const optionsContainer = document.getElementById('options-container');
    const progressFill = document.getElementById('progress-fill');
    const explanationArea = document.getElementById('explanation-area');
    const nextBtn = document.getElementById('next-btn');

    function shuffle(array) {
        for (let i = array.length - 1; i > 0; i--) {
            const j = Math.floor(Math.random() * (i + 1));
            [array[i], array[j]] = [array[j], array[i]];
        }
        return array;
    }

    function startQuiz(mode) {
        // Get selected difficulty
        const diffRadios = document.getElementsByName('difficulty');
        let selectedDiff = "中級";
        for(const r of diffRadios) if(r.checked) selectedDiff = r.value;

        // Filter
        let filtered = questionDatabase.filter(q => q.difficulty === selectedDiff);
        if (mode !== 'mixed') {
            filtered = filtered.filter(q => q.genre === mode);
        }

        if (filtered.length < MAX_QUESTIONS) {
            alert(`申し訳ありません。「${mode}」の「${selectedDiff}」問題は${MAX_QUESTIONS}問に満たないため、開始できません。他の組み合わせをお試しください。（現在 ${filtered.length} 問）`);
            return;
        }

        // Setup
        shuffle(filtered);
        currentQuestions = filtered.slice(0, MAX_QUESTIONS);

        currentQuestionIndex = 0;
        score = 0;
        userMistakes = [];

        startScreen.classList.remove('active');
        resultScreen.classList.remove('active');
        quizScreen.classList.add('active');

        renderQuestion();
    }

    function renderQuestion() {
        const q = currentQuestions[currentQuestionIndex];
        // UI
        questionText.textContent = q.question;
        genreBadge.textContent = q.genre;
        diffBadge.textContent = q.difficulty;
        diffBadge.className = `badge badge-diff ${q.difficulty}`; // 色変え用クラス

        document.getElementById('question-counter').textContent = `Q ${currentQuestionIndex + 1} / ${currentQuestions.length}`;
        document.getElementById('score-counter').textContent = `Score: ${score}`;

        const percent = ((currentQuestionIndex) / currentQuestions.length) * 100;
        progressFill.style.width = `${percent}%`;

        explanationArea.style.display = 'none';
        nextBtn.style.display = 'none';
        optionsContainer.innerHTML = '';

        // Options
        const optionsToShow = [...q.options];
        // shuffle(optionsToShow); // 選択肢ランダム化する場合はコメントアウト解除

        optionsToShow.forEach((opt) => {
            const btn = document.createElement('button');
            btn.className = 'option-btn';
            btn.textContent = opt.text;
            btn.onclick = () => handleAnswer(opt, btn, q);
            optionsContainer.appendChild(btn);
        });
    }

    function handleAnswer(selectedOption, btnElement, questionObj) {
        // Lock UI
        const allBtns = optionsContainer.querySelectorAll('.option-btn');
        allBtns.forEach(b => b.disabled = true);
        nextBtn.style.display = 'block';

        const isCorrect = selectedOption.isCorrect;
        if (isCorrect) {
            btnElement.classList.add('correct');
            score++;
            document.getElementById('result-icon').textContent = "⭕";
            document.getElementById('result-text').textContent = "正解！";
            document.getElementById('result-text').style.color = "var(--success)";
        } else {
            btnElement.classList.add('incorrect');
            allBtns.forEach(b => {
                if (b.textContent === questionObj.options.find(o => o.isCorrect).text) {
                    b.classList.add('correct');
                }
            });
            document.getElementById('result-icon').textContent = "❌";
            document.getElementById('result-text').textContent = "不正解...";
            document.getElementById('result-text').style.color = "var(--error)";

            userMistakes.push({
                question: questionObj.question,
                correctAnswer: questionObj.options.find(o => o.isCorrect).text,
                explanation: questionObj.mainExplanation
            });
        }
        document.getElementById('score-counter').textContent = `Score: ${score}`;

        // Explanation & Links
        document.getElementById('main-explanation').textContent = questionObj.mainExplanation;

        // Build Google Search Links
        const refContainer = document.getElementById('ref-links');
        refContainer.innerHTML = '';
        if (questionObj.keywords && questionObj.keywords.length > 0) {
            questionObj.keywords.forEach(kw => {
                const link = document.createElement('a');
                link.className = 'ref-link';
                // 検索URL生成
                link.href = `https://www.google.com/search?q=${encodeURIComponent(kw)}`;
                link.target = "_blank";
                link.innerHTML = `🔍 ${kw}`;
                refContainer.appendChild(link);
            });
        } else {
            refContainer.textContent = "関連キーワードなし";
        }

        // Option Feedback
        const fbContainer = document.getElementById('option-feedbacks');
        fbContainer.innerHTML = '';
        questionObj.options.forEach((opt) => {
            const item = document.createElement('div');
            item.className = 'option-fb-item';
            const marker = opt.isCorrect ? "○" : "×";
            const color = opt.isCorrect ? "var(--success)" : "var(--error)";
            item.innerHTML = `<span class="fb-marker" style="color:${color}">${marker}</span> <b>${opt.text}</b><br><span style="font-size:0.85rem; color:#666; margin-left:28px; display:block; margin-top:2px;">👉 ${opt.feedback}</span>`;
            fbContainer.appendChild(item);
        });

        explanationArea.style.display = 'block';
        explanationArea.scrollIntoView({ behavior: 'smooth', block: 'center' });
    }

    function nextQuestion() {
        currentQuestionIndex++;
        if (currentQuestionIndex < currentQuestions.length) {
            renderQuestion();
        } else {
            showResults();
        }
    }

    function showResults() {
        quizScreen.classList.remove('active');
        resultScreen.classList.add('active');
        document.getElementById('final-score').textContent = score;
        document.getElementById('total-count').textContent = currentQuestions.length;

        const msg = document.getElementById('feedback-message');
        const percentage = (score / currentQuestions.length) * 100;

        if (percentage === 100) {
            msg.textContent = "素晴らしい！完璧です！🎉";
            msg.style.color = "var(--primary)";
        } else if (percentage >= 80) {
            msg.textContent = "合格圏内です！その調子！👍";
            msg.style.color = "var(--secondary)";
} else {
            msg.textContent = "復習して弱点を克服しましょう！💪";
            msg.style.color = "var(--text)";
        }

        const list = document.getElementById('review-list');
        list.innerHTML = '';
        if (userMistakes.length === 0) {
            list.innerHTML = '<li style="text-align:center; list-style:none; padding:20px;">全問正解です！復習事項はありません✨</li>';
        } else {
            userMistakes.forEach(mistake => {
                const li = document.createElement('li');
                li.className = 'review-item';
                li.innerHTML = `
                    <div class="review-q">Q. ${mistake.question}</div>
                    <div class="review-ans">正解: ${mistake.correctAnswer}</div>
                    <div style="font-size:0.9rem; color:#555;">${mistake.explanation}</div>
                `;
                list.appendChild(li);
            });
        }
    }
